Languages#

Iterable formatting#

var iterableFormat = new IterableFormat(locale: "{{selectedLocale}}");
iterableFormat.format([]) "{{formatCount(0)}}"
iterableFormat.format([1]) -> "{{formatCount(1)}}"
iterableFormat.format([1, 2]) -> "{{formatCount(2)}}"
iterableFormat.format([1, 2, 3]) -> "{{formatCount(3)}}"

Relative time formatting#

var durationFormat = new DurationFormat(locale: "{{selectedLocale}}");
var defaultAgeFormat = new AgeFormat(locale: "{{selectedLocale}}");
var secondsAgeFormat = new AgeFormat(locale: "{{selectedLocale}}", rounder: new DurationRounder.staticUnit(TimeUnit.SECOND));
var duration = new Duration(: );
durationFormat.format(duration) {{duration}}
